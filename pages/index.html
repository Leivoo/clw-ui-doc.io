# npm 安装

使用 npm 或 yarn 的方式安装。

npm install clw-ui
yarn add clw-ui

npm install element-ui
yarn add element-ui

# 快速上手

import CLWUI from "clw-ui";
import "clw-ui/lib/clw-ui.css";

import ElementUI from "element-ui";
import "element-ui/lib/theme-chalk/index.css";

Vue.use(CLWUI);
Vue.use(ElementUI);
